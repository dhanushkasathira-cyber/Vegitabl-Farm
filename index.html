        function updateMetrics() {

            for (let key in metrics) {

                const metric = metrics[key];

                const variation = (Math.random() - 0.5) * (metric.max - metric.min) * 0.1;

                let newValue = metric.current + variation;

                

                // Keep within bounds

                newValue = Math.max(metric.min, Math.min(metric.max, newValue));

                

                if (key === 'phLevel') {

                    newValue = Math.round(newValue * 10) / 10;

                } else {

                    newValue = Math.round(newValue);

                }

                

                metric.current = newValue;

                

                const element = document.getElementById(key);

                if (element) {

                    element.textContent = newValue;

                    

                    // Update status color based on optimal range

                    const statusEl = element.parentElement.querySelector('.metric-status');

                    const midPoint = (metric.min + metric.max) / 2;

                    const range = metric.max - metric.min;

                    

                    if (Math.abs(newValue - midPoint) < range * 0.2) {

                        statusEl.className = 'metric-status status-optimal';

                        statusEl.textContent = 'Optimal';

                    } else {

                        statusEl.className = 'metric-status status-warning';

                        statusEl.textContent = 'Adjusting';

                    }

                }

            }

            

            // Update last updated time

            const now = new Date();

            document.getElementById('lastUpdate').textContent = now.toLocaleTimeString();

            

            updateChart();

        }
